
// 2Dゲームで用いられるスプライトを表すクラス
class Sprite
{
   頂点バッファ
   インデックスバッファ

   テクスチャ
   テクスチャ転送元矩形
   ピボット
   1単位あたりのピクセルする
};


// 全てのコンポーネントの基底クラス
class Component
{
   ゲームオブジェクトへの参照

   更新処理( 時間 );
   描画処理( );
};


// アフィン変換情報と親子関係を扱うクラス
class Transform : Component
{
   スケール
   向き
   位置
   ワールド変換行列
   ワールド変換行列の逆行列
};


// スプライトのレンダリング機能
class SpriteRenderer
{
   マテリアルへの参照
   スプライトへの参照
   頂点カラー
   フリップ
};


class GameObject
{
   コンポーネントのリスト;
};



class Scene
{
    ゲームオブジェクトのリスト
};


class SceneManager
{
    アクティブなシーンのリスト
    全てのシーンのリスト
};
